            function getCookie(cname) {
                var name = cname + "=";
                var decodedCookie = decodeURIComponent(document.cookie);
                var ca = decodedCookie.split(';');
                for (var i = 0; i < ca.length; i++) {
                    var c = ca[i];
                    while (c.charAt(0) == ' ') {
                        c = c.substring(1);
                    }
                    if (c.indexOf(name) == 0) {
                        return c.substring(name.length, c.length);
                    }
                }
                return "";
            }
            
            var rand_num = Date.now()%100
            var expires = new Date(Date.now() + 31536000000);
            
            if (!getCookie("ab-checking")){
                document.cookie = "ab-checking=" + rand_num +"; expires=" + expires + "; path=/";
            }
            if (!getCookie("ab-testing") && getCookie("ab-checking") <= 0) { 
                document.cookie = "ab-testing=true; expires=" + expires + "; path=/";
